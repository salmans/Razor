Truth =>   (exists x y . Q(x,y))
Q(x,y) =>  (exists x . Q1(x,y) ) | (exists x . Q2(x,y))


Q1(x,y) =>  (exists x . Q11(x,y) ) | (exists x . Q12(x,y))
Q2(x,y) =>  (exists x . Q21(x,y) ) | (exists x . Q22(x,y))

Q21(x,y) =>  (exists x . Q211(x,y) ) | (exists x . Q212(x,y))
Q22(x,y) =>  (exists x . Q221(x,y) ) | (exists x . Q222(x,y))
